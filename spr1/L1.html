import random
import math
import matplotlib.pyplot as plt
import numpy as np

# Funkcja do obliczania wspczynnika Newtona (n choose k)
def binomial_coeff(n, k):
    return math.comb(n, k)

# Funkcja do teoretycznej gstoci rozkadu dwumianowego
def binomial_pdf(n, p, k):
    return binomial_coeff(n, k) * (p ** k) * ((1 - p) ** (n - k))

# Funkcja do empirycznej gstoci
def empirical_pdf(n, p, M):
    successes_count = [0] * (n + 1)
    
    for _ in range(M):
        successes = binomial_trial(n, p)
        successes_count[successes] += 1
    
    return [count / M for count in successes_count]

# Funkcja do symulacji prby z rozkadu dwumianowego
def binomial_trial(n, p):
    successes = 0
    for _ in range(n):
        if random.random() < p:
            successes += 1
    return successes

# Funkcja do teoretycznej dystrybuanty
def binomial_cdf(n, p, k):
    return sum(binomial_pdf(n, p, i) for i in range(k + 1))

# Funkcja do empirycznej dystrybuanty
def empirical_cdf(n, p, M):
    successes_count = [0] * (n + 1)
    
    for _ in range(M):
        successes = binomial_trial(n, p)
        successes_count[successes] += 1
    
    cumulative_count = np.cumsum(successes_count)
    return cumulative_count / M

# Parametry
n = 100  # liczba prb
p = 0.5  # prawdopodobiestwo sukcesu

# Funkcja do obliczania wspczynnika Newtona (n choose k)
def binomial_coeff(n, k):
    return math.comb(n, k)

# Funkcja do teoretycznej gstoci rozkadu dwumianowego
def binomial_pdf(n, p_vector, k):
    # Prawdopodobiestwo sukcesu w danej prbie
    # Obliczamy prawdopodobiestwo sukcesu dla k prb
    probability = 1
    for i in range(k):
        probability *= p_vector[i]  # Produkt prawdopodobiestw sukcesw
    for i in range(k, n):
        probability *= (1 - p_vector[i])  # Produkt prawdopodobiestw poraek
    return binomial_coeff(n, k) * probability

# Funkcja do empirycznej gstoci
def empirical_pdf(n, p_vector, M):
    successes_count = [0] * (n + 1)
    
    for _ in range(M):
        successes = sum(binomial_vector_with_p(n, p_vector))
        successes_count[successes] += 1
    
    return [count / M for count in successes_count]

# Funkcja do symulacji prby z rozkadu dwumianowego z wektorem p
def binomial_vector_with_p(n, p_vector):
    return [1 if random.random() < p_vector[i] else 0 for i in range(n)]

# Funkcja do teoretycznej dystrybuanty
def binomial_cdf(n, p_vector, k):
    return sum(binomial_pdf(n, p_vector, i) for i in range(k + 1))

def clopper_pearson(successes, trials, alpha=0.05): #liczba sukcesw, liczba prb, poziom ufnoci
    beta_ppf = scipy.stats.beta.ppf #punktowa funkcja procentowa zmiennej beta
    low = beta_ppf(alpha / 2, successes, trials - successes + 1) #dolna granica
    high = beta_ppf(1 - alpha / 2, successes + 1, trials - successes) # grna granica
    
    if math.isnan(low): 
        low = 0
    if math.isnan(high):
        high = 1
    
    return [low, high]

Gsto rozkadu dwumianowego opisujemy za pomoc wzoru

$$ P(N) = \binom{n}{N}p^N(1-p)^{n-N},$$

gdzie $n$ oznacza liczb prb, $N$ liczb sukcesw oraz $p$ prawdopodobietwo sukcesu.

Aby generowa prbki z rozkadu dwumianowego w Pythonie, moemy uy biblioteki NumPy.

In[401]:
n = 100 #liczba prb
N = np.linspace(0,n, 101, endpoint=True) #0,1,2,...,100
p = 1/3 #prawdopodobietwo sukcesu

theoretical_density = np.zeros(n)
for i in range(n):
    theoretical_density[i] = scipy.special.binom(n, N[i])*p**N[i]*(1-p)**(n-N[i]) #teoretyczna gsto wyznaczona ze wzoru powyej

samples = np.random.binomial(n, p, 100000) #wygenerowanie duej liczby prbek

plt.hist(samples, density=True, label = "rozkad prbki", bins=20)
plt.plot(theoretical_density, label = "teoretyczna gsto")
plt.grid(alpha=0.3)
plt.title("Porwnanie histogramu wygenerowanych prbek z teoretyczn gstoci rozkadu dwumianowego")
plt.legend()
